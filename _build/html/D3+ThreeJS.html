D3.js + Three.js ‚Üí Aero Adobe¬∂

We are looking at connecting D3.js + Three.js. A common question I would receive when leading discussions or courses would be

‚ÄúCan I create 3D graphs with D3.js?‚Äù

My response would be

‚ÄúD3 is only 2D, to do 3D you need to use Three, but you can use D3 in Three.‚Äù

Partly as a joke, but mostly because its true. So, this is what this notebook is trying to showcase, coupling d3.js and three.js. Remember, D3 stands for Data Driven Documents and it does an excellent job of creating a means to format data to be used in three.js.

The output for this will be a .gltf file or a GL Transmission Format file, which is used in blender or Adobe Aero.

Even though we will be using three.js we will be using the three.js modules. So, as always‚Ä¶

Our Data¬∂

For our project, we are going to create a 3D cube and with values from a 2D normal distribution. To create our data, we can use

d3.randomNormal(mu, sigma)

the mu value or the mean (Œº)

and

the sigma is the standard deviation (œÉ)

Let‚Äôs create a 2D version of this only in d3.js.

Generating the Values (and Sorting)¬∂

with Œº = .5 and œÉ = .1

Histogram of the Random Normal Distribution¬∂

All d3.randomNormal() is doing is producing a random value based on the Œº and œÉ provide. The bell curve (that we are used to seeing) is a histogram of those values. Meaning, if we are given

3,4,4,4,5,5,5,6,6,7

We could produce the following table of data

d3.rollup(data, values to rollup, count of values)

<script>
    var sigmahist = .1
    var meanhist = .5
    var data = d3.range(11).map(d=>0)
    var points = d3.range(100).map(d=> Math.round(d3.randomNormal(meanhist, sigmahist)(d)*10))
    var rolledData = d3.rollup(points,v => v.length, d => d)
    
    for (let [key, value] of rolledData) {
        data[key] = value
    }
    var widthHist = 400
    var heightHist = 300
    var marginHist = 40 
    var svg = d3.select("div#histogram").append("svg")
        .attr("width", widthHist)
        .attr("height", heightHist)            

    var xHist = d3.scaleLinear().range([marginHist, widthHist-marginHist]).domain(d3.extent(d3.range(11)))
    var yHist = d3.scaleLinear().range([heightHist-marginHist, marginHist]).domain([0,50])
    
    var palette = d3.interpolateReds
    var color = d3.scaleLinear().range([0,1]).domain([0,50])

    var xAxisHist = d3.axisBottom().scale(xHist)
    svg.append("g")
        .attr("class", "xaxis")
        .attr("transform", "translate(0," + (heightHist-marginHist) + ")")
        .call(xAxisHist) 
        
    svg.append("text")
        .attr("x", widthHist/2)
        .attr("y", heightHist-5)
        .style("text-anchor", "middle")
        .text("Indices")
        
    var yAxisHist = d3.axisLeft().scale(yHist)
    svg.append("g")
        .attr("class", "yaxis")
        .attr("transform", "translate(" + marginHist + ",0)")
        .call(yAxisHist)
        
    svg.append("text")
        .attr("transform", "rotate(-90,15,"+(heightHist/2)+")")
        .attr("x", 15)
        .attr("y", heightHist/2)
        .style("text-anchor", "middle")
        .text("Count")
        
    var radius = 20
    svg.selectAll("rect").data(data)
        .join("rect")
        .attr("class", "hist")
        .attr("x",(d,i)=> xHist(i)-(radius/2))
        .attr("y",(d,i)=> yHist(d)-(radius/2))
        .attr("rx",radius/4)
        .attr("width", radius)
        .attr("height", radius)
        .style("stroke","darkgrey")
        .style("fill",(d,i)=> palette(color(d)))

    function updateSigmaHist(s){
        sigmahist = +s/20
        d3.select("p#sigmahist").text(sigmahist)
        updateGraphHist()
    }
    
    function updateMuHist(m){
        meanhist = +m/10
        d3.select("p#meanhist").text(meanhist)
        updateGraphHist()
    }
    
    function updateGraphHist() {
        var points = d3.range(100).map(d=> Math.round(d3.randomNormal(meanhist, sigmahist)(d)*10))
        var rolledData = d3.rollup(points,v => v.length, d => d)
        
        data = d3.range(11).map(d=> 0)
        for (let [key, value] of rolledData) {
            data[key] = value
        }
        data = data.filter((d,i) => (i>=0) && (i<=10))
        console.log(data)
        d3.selectAll("rect.hist").data(data)
            .transition()
            .attr("y",(d,i)=> yHist(d)-(radius/2))
            .style("fill",(d,i)=> palette(color(d)))
    }
</script>

<script type="module">
    import * as THREE from 'https://threejs.org/build/three.module.js';
    import * as d3 from "https://cdn.skypack.dev/d3@7";
    import { OrbitControls } from 'https://threejs.org/examples/jsm/controls/OrbitControls.js';

    var squares = 10
    var sigma = 100
    var mu = (squares*squares*squares)/2
    var points = d3.range(10000).map(d=> d3.randomNormal(mu, sigma)(d))
    points = points.map(d=> Math.round(d))
    var data = d3.range(squares*squares*squares).map(d=>0)
    var rolledData = d3.rollup(points,v => v.length, d => d)
    for (let [key, value] of rolledData) {
        data[key] = value
    }
    
    var height = 400
    var width = 400

    var palette = d3.interpolateReds
    var color = d3.scaleLinear().domain(d3.extent(data)).range([0,1])

    let renderer = new THREE.WebGLRenderer({antialias: true});
    renderer.setSize(width, height);
    renderer.setPixelRatio(devicePixelRatio);
    var scene = new THREE.Scene();
    var camera = new THREE.PerspectiveCamera(
        45, // Field of view
        height / width, // Aspect ratio
        0.1, // Near
        3000 // Far
    );

    var controls = new OrbitControls(camera, renderer.domElement)

    controls.addEventListener("change", () => renderer.render(scene, camera))
    camera.position.x = 0;
    camera.position.y = 0;
    camera.position.z = 1000;
    var cubes = new THREE.Object3D();
    scene.add( cubes );
    var x = d3.scaleLinear().range([-width/2,width/2]).domain([0,squares-1])
    var y = d3.scaleLinear().range([-width/2,width/2]).domain([0,squares-1])
    var z = d3.scaleLinear().range([-width/2,width/2]).domain([0,squares-1])
    var counter = 0;
    console.log(data)

    d3.range(0,squares).forEach(function(dx) {
        d3.range(0,squares).forEach(function(dy) {
            d3.range(0,squares).forEach(function(dz) {
                //console.log(data[counter])
                if (data[counter]>10) {
                    var size = width/(squares-1)
                    var geometry = new THREE.BoxGeometry(size,size,size);
                    var material = new THREE.MeshBasicMaterial({color: palette(color(data[counter])), transparent : true, opacity:color(data[counter])})
                    var cube = new THREE.Mesh(geometry, material);
                    cube.position.x = x(dx);
                    cube.position.y = y(dy);
                    cube.position.z = z(dz);
                    cubes.add(cube);
